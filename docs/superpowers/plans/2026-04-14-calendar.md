# Calendar Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a 📅 Calendar tab to the TC Plants lab notes app that shows past experiment dates and upcoming review dates on a month grid, with a slide-in day-detail panel and print/PDF export for a week or month.

**Architecture:** All code lives inside `index.html` (single-file SPA). The calendar reads from in-memory `S.notes` — no new API calls or encrypted files. A `_buildCalMap()` helper pre-indexes notes by date. The day-detail panel uses the same slide-in overlay pattern as `#history-panel`. Print export follows the same `window.open()` pattern as `printNote()`.

**Tech Stack:** Vanilla JS, CSS custom properties (existing), `marked.js` (already loaded), no new dependencies.

function _buildCalMap(){
  // Returns: { 'YYYY-MM-DD': [{id, title, type, stage, species, dotType, isReview}] }
  const map={};
  const add=(d,entry)=>{
    if(!d) return;
    const key=d.slice(0,10);
    if(!map[key]) map[key]=[];
    map[key].push(entry);
  };
  for(const n of S.notes){
    // Experiment date — blue dot (red if contamination)
    add(n.date,{
      id:n.id, title:n.title, type:n.type||'', stage:n.stage||'', species:n.species||'',
      dotType: n.type==='Contamination Log'?'red':'blue', isReview:false
    });
    // Review date — orange dot
    if(n.next_review){
      add(n.next_review,{
        id:n.id, title:n.title, type:n.type||'', stage:n.stage||'', species:n.species||'',
        dotType:'orange', isReview:true
      });
    }
  }
  return map;
}
```

function renderCalendar(){
  const y=S.calYear, m=S.calMonth;
  const map=_buildCalMap();
  const today=new Date().toISOString().slice(0,10);

  // Month label
  const label=new Date(y,m,1).toLocaleDateString('en-IN',{month:'long',year:'numeric'});
  document.getElementById('cal-month-label').textContent=label;

  // Day-of-week headers + cells
  const dows=['Mon','Tue','Wed','Thu','Fri','Sat','Sun'];
  let html=dows.map(d=>`<div class="cal-dow">${d}</div>`).join('');

  // First day of month (0=Sun..6=Sat); we want Mon-start grid
  const firstDay=new Date(y,m,1).getDay(); // 0=Sun
  const startOffset=(firstDay===0)?6:firstDay-1; // blanks before day 1
  const daysInMonth=new Date(y,m+1,0).getDate();
  const prevMonthDays=new Date(y,m,0).getDate();

  // Prev month info (JS Date handles year rollback automatically)
  const prevMonthDate=new Date(y,m,0);
  const prevY=prevMonthDate.getFullYear();
  const prevM=String(prevMonthDate.getMonth()+1).padStart(2,'0');

  // Next month info (JS Date handles year rollforward automatically)
  const nextMonthDate=new Date(y,m+1,1);
  const nextY=nextMonthDate.getFullYear();
  const nextM=String(nextMonthDate.getMonth()+1).padStart(2,'0');

  // Cells before month start (dimmed, previous month)
  for(let i=startOffset-1;i>=0;i--){
    const d=prevMonthDays-i;
    const dateStr=`${prevY}-${prevM}-${String(d).padStart(2,'0')}`;
    html+=_calCell(dateStr,d,map,today,true);
  }

  // Current month cells
  const curM=String(m+1).padStart(2,'0');
  for(let d=1;d<=daysInMonth;d++){
    const dateStr=`${y}-${curM}-${String(d).padStart(2,'0')}`;
    html+=_calCell(dateStr,d,map,today,false);
  }

  // Trailing cells to complete the last row
  const total=startOffset+daysInMonth;
  const trailing=(7-(total%7))%7;
  for(let d=1;d<=trailing;d++){
    const dateStr=`${nextY}-${nextM}-${String(d).padStart(2,'0')}`;
    html+=_calCell(dateStr,d,map,today,true);
  }

  document.getElementById('cal-grid').innerHTML=html;

  // Re-open panel if a date was selected
  if(S.calSelDate) openCalendarDay(S.calSelDate);
}

function _calCell(dateStr,dayNum,map,today,dim){
  const entries=map[dateStr]||[];
  const isToday=dateStr===today;
  const isSelected=dateStr===S.calSelDate;
  let cls='cal-day';
  if(dim) cls+=' dim';
  if(isToday) cls+=' today';
  if(isSelected) cls+=' selected';

  // Build dots (max 3, then +N)
  let dots='';
  const show=entries.slice(0,3);
  show.forEach(e=>{ dots+=`<span class="cal-dot ${e.dotType}"></span>`; });
  if(entries.length>3) dots+=`<span class="cal-dot-more">+${entries.length-3}</span>`;

  return `<div class="${cls}" onclick="openCalendarDay('${dateStr}')">
    <div class="cal-day-num">${dayNum}</div>
    <div class="cal-dots">${dots}</div>
  </div>`;
}

function _calPrev(){
  if(S.calMonth===0){S.calMonth=11;S.calYear--;}else{S.calMonth--;}
  S.calSelDate=null;
  _closeCalPanel();
  renderCalendar();
}
function _calNext(){
  if(S.calMonth===11){S.calMonth=0;S.calYear++;}else{S.calMonth++;}
  S.calSelDate=null;
  _closeCalPanel();
  renderCalendar();
}
function _calToday(){
  const now=new Date();
  S.calYear=now.getFullYear();
  S.calMonth=now.getMonth();
  S.calSelDate=null;
  _closeCalPanel();
  renderCalendar();
}
function _calExportMenu(){
  document.getElementById('cal-export-menu').classList.toggle('on');
}
// Close export menu when clicking outside
document.addEventListener('click',function(e){
  if(!e.target.closest('.cal-export-wrap')){
    document.getElementById('cal-export-menu')?.classList.remove('on');
  }
});
```

async function exportCalendar(scope){
  document.getElementById('cal-export-menu').classList.remove('on');
  const today=new Date();

  // Compute date range
  let startDate, endDate;
  if(scope==='week'){
    const dow=today.getDay(); // 0=Sun
    const monday=new Date(today);
    monday.setDate(today.getDate()-((dow+6)%7));
    const sunday=new Date(monday);
    sunday.setDate(monday.getDate()+6);
    startDate=monday.toISOString().slice(0,10);
    endDate=sunday.toISOString().slice(0,10);
  } else {
    startDate=`${today.getFullYear()}-${String(today.getMonth()+1).padStart(2,'0')}-01`;
    const lastDay=new Date(today.getFullYear(),today.getMonth()+1,0).getDate();
    endDate=`${today.getFullYear()}-${String(today.getMonth()+1).padStart(2,'0')}-${String(lastDay).padStart(2,'0')}`;
  }

  // Collect all entries in range
  const map=_buildCalMap();
  const inRange=Object.entries(map)
    .filter(([d])=>d>=startDate&&d<=endDate)
    .sort(([a],[b])=>a<b?-1:1);

  if(inRange.length===0){
    toast('No entries in this period','warn'); return;
  }

  // Show preparing indicator
  toast('Preparing export…','info');

  // Build print sections — decrypt bodies
  let sections='';
  for(const [dateStr,entries] of inRange){
    const d=new Date(dateStr+'T00:00:00');
    const dateLabel=d.toLocaleDateString('en-IN',{weekday:'long',day:'numeric',month:'long',year:'numeric'});
    sections+=`<h2>${dateLabel}</h2>`;

    // Deduplicate by id (a note can appear twice if experiment + review on same day)
    const seen=new Set();
    for(const e of entries){
      if(seen.has(e.id)) continue;
      seen.add(e.id);
      const meta=S.notes.find(n=>n.id===e.id);
      if(!meta) continue;

      // Fetch body (cache-first)
      let body=S.bodyCache[e.id]||null;
      if(!body){
        try{ const r=await loadNote(e.id); body=r.body; S.bodyCache[e.id]=body; }catch(err){ body=''; }
      }

      const isContam=meta.type==='Contamination Log';
      const isReview=!!meta.next_review&&meta.next_review.slice(0,10)===dateStr;
      const flags=[];
      if(isContam) flags.push('<span class="flag contam">☣ Contamination</span>');
      if(isReview) flags.push('<span class="flag review">⏰ Review Due</span>');

      sections+=`<div class="entry">
        <h3>${esc(meta.title)}${flags.join('')}</h3>
        <div class="meta-row">
          ${meta.type?`<span class="badge">${esc(meta.type)}</span>`:''}
          ${meta.stage?`<span class="badge">${esc(meta.stage)}</span>`:''}
          ${meta.species?`<span class="badge">${esc(meta.species)}</span>`:''}
          ${meta.tags?.length?`<span class="badge">🏷 ${esc(meta.tags.join(', '))}</span>`:''}
          ${meta.recipe_id&&S.recs.find(r=>r.id===meta.recipe_id)?`<span class="badge">🧪 ${esc(S.recs.find(r=>r.id===meta.recipe_id).name)}</span>`:''}
        </div>
        ${body?`<div class="body-content">${marked.parse(body)}</div>`:'<p class="no-body"><em>(no body)</em></p>'}
      </div>`;
    }
  }
