// ── SESSION CACHE ────────────────────────────────────────────
function _writeCache(){
  sessionStorage.setItem('tcplants_cache', JSON.stringify({
    notes:S.notes, accs:S.accs, recs:S.recs,
    notesSha:S.notesSha, accSha:S.accSha, recSha:S.recSha,
    cachedAt:Date.now()
  }));
}
function _readCache(){
  try{
    const raw=sessionStorage.getItem('tcplants_cache');
    if(!raw) return null;
    const c=JSON.parse(raw);
    if(Date.now()-c.cachedAt>600000) return null; // 10 min TTL
    return c;
  }catch(e){ return null; }
}

// ── WEATHER ──────────────────────────────────────────────────
const WX_LOCS = {
  lab:{lat:12.7434,lon:80.1931,label:'Lab'},
  gh: {lat:12.7742,lon:80.2105,label:'Greenhouse'}
};
async function fetchWeather(){
  const base='https://api.open-meteo.com/v1/forecast?current=apparent_temperature,relative_humidity_2m,uv_index,precipitation_probability&timezone=Asia%2FKolkata';
  try{
    const [rLab,rGh]=await Promise.all([
      fetch(`${base}&latitude=${WX_LOCS.lab.lat}&longitude=${WX_LOCS.lab.lon}`).then(r=>r.json()),
      fetch(`${base}&latitude=${WX_LOCS.gh.lat}&longitude=${WX_LOCS.gh.lon}`).then(r=>r.json())
    ]);
    S.weather.lab=rLab.current;
    S.weather.gh=rGh.current;
    S.weather.updatedAt=new Date();
    renderWeatherChip();
    renderWeatherPop();
  }catch(e){ /* silent fail */ }
}
function renderWeatherChip(){
  const d=S.weather.lab; if(!d) return;
  const chip=document.getElementById('weather-chip');
  chip.classList.add('on');
  chip.innerHTML=`🌡️ ${Math.round(d.apparent_temperature)}°C`;
}
function renderWeatherPop(){
  const d=S.weather; if(!d.lab||!d.gh) return;
  const fmt=(loc,key)=>{
    const c=d[loc];
    document.getElementById(`wx-${loc}-feels`).textContent=`Feels ${Math.round(c.apparent_temperature)}°C`;
    document.getElementById(`wx-${loc}-hum`).textContent=`${c.relative_humidity_2m}% humidity`;
    document.getElementById(`wx-${loc}-uv`).textContent=`UV Index ${c.uv_index}`;
    document.getElementById(`wx-${loc}-rain`).textContent=`${c.precipitation_probability}% rain`;
  };
  fmt('lab'); fmt('gh');
  const t=d.updatedAt;
  document.getElementById('wx-updated').textContent=`Updated ${t.toLocaleDateString('en-IN',{day:'numeric',month:'short'})} ${t.toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'})}`;
}
```

// ── DRAFT AUTO-SAVE ──────────────────────────────────────────
let _draftTimer=null;
function _saveDraft(){
  const d={
    editing:S.editing||null,
    title:document.getElementById('ne-title').value,
    type:document.getElementById('ne-type').value,
    stage:document.getElementById('ne-stage').value,
    species:document.getElementById('ne-species').value,
    tags:document.getElementById('ne-tags').value,
    review:document.getElementById('ne-review').value,
    recipe_id:document.getElementById('ne-recipe').value,
    cultures_out:document.getElementById('ne-cultures-out').value,
    body:document.getElementById('ne-body').value,
    savedAt:Date.now()
  };
  sessionStorage.setItem('tcplants_draft',JSON.stringify(d));
}
function _clearDraft(){ sessionStorage.removeItem('tcplants_draft'); document.getElementById('draft-banner').classList.remove('on'); }
function _checkDraft(editingId){
  try{
    const raw=sessionStorage.getItem('tcplants_draft');
    if(!raw) return;
    const d=JSON.parse(raw);
    if(d.editing!==editingId) return; // wrong context
    // restore
    document.getElementById('ne-title').value=d.title||'';
    document.getElementById('ne-type').value=d.type||'';
    document.getElementById('ne-stage').value=d.stage||'';
    document.getElementById('ne-species').value=d.species||'';
    document.getElementById('ne-tags').value=d.tags||'';
    document.getElementById('ne-review').value=d.review||'';
    document.getElementById('ne-recipe').value=d.recipe_id||'';
    document.getElementById('ne-cultures-out').value=d.cultures_out||'';
    document.getElementById('ne-body').value=d.body||'';
    const ago=Math.round((Date.now()-d.savedAt)/60000);
    document.getElementById('draft-msg').textContent=`Draft restored from ${ago<1?'just now':ago+'m ago'} — unsaved changes.`;
    document.getElementById('draft-banner').classList.add('on');
  }catch(e){}
}
function _wireDraftListeners(){
  const ids=['ne-title','ne-type','ne-stage','ne-species','ne-tags','ne-review','ne-recipe','ne-cultures-out','ne-body'];
  ids.forEach(id=>{
    const el=document.getElementById(id);
    el.addEventListener('input',()=>{ clearTimeout(_draftTimer); _draftTimer=setTimeout(_saveDraft,1500); });
    el.addEventListener('change',()=>{ clearTimeout(_draftTimer); _draftTimer=setTimeout(_saveDraft,1500); });
  });
  document.getElementById('draft-discard-btn').addEventListener('click',()=>{
    _clearDraft();
    S.editing ? startEditNote() : startNewNote();
  });
}
```

// ── SVG CHART HELPERS ────────────────────────────────────────
function svgLine(points, w, h, color='var(--ac)', fill=false){
  if(points.length<2) return `<svg width="${w}" height="${h}"></svg>`;
  const xs=points.map(p=>p[0]), ys=points.map(p=>p[1]);
  const minX=Math.min(...xs),maxX=Math.max(...xs),minY=Math.min(...ys),maxY=Math.max(...ys);
  const px=(x)=>((x-minX)/(maxX-minX||1))*(w-20)+10;
  const py=(y)=>(h-10)-((y-minY)/(maxY-minY||1))*(h-20);
  const pts=points.map(p=>`${px(p[0]).toFixed(1)},${py(p[1]).toFixed(1)}`).join(' ');
  const area=fill?`<polygon points="${pts} ${px(maxX).toFixed(1)},${h-10} ${px(minX).toFixed(1)},${h-10}" fill="${color}" opacity=".15"/>`:'' ;
  return `<svg width="${w}" height="${h}" viewBox="0 0 ${w} ${h}" style="overflow:visible">
    ${area}<polyline points="${pts}" fill="none" stroke="${color}" stroke-width="2" stroke-linejoin="round"/>
    ${points.map(p=>`<circle cx="${px(p[0]).toFixed(1)}" cy="${py(p[1]).toFixed(1)}" r="3" fill="${color}"/>`).join('')}
  </svg>`;
}
function svgDonut(slices, size=120){
  const total=slices.reduce((s,sl)=>s+sl.v,0)||1;
  let angle=-Math.PI/2, r=size/2, ri=r*0.55, cx=r, cy=r;
  const paths=slices.map(sl=>{
    const a=(sl.v/total)*Math.PI*2;
    const x1=cx+r*Math.cos(angle), y1=cy+r*Math.sin(angle);
    const x2=cx+r*Math.cos(angle+a), y2=cy+r*Math.sin(angle+a);
    const xi1=cx+ri*Math.cos(angle), yi1=cy+ri*Math.sin(angle);
    const xi2=cx+ri*Math.cos(angle+a), yi2=cy+ri*Math.sin(angle+a);
    const large=a>Math.PI?1:0;
    const d=`M${xi1.toFixed(1)},${yi1.toFixed(1)} L${x1.toFixed(1)},${y1.toFixed(1)} A${r},${r} 0 ${large} 1 ${x2.toFixed(1)},${y2.toFixed(1)} L${xi2.toFixed(1)},${yi2.toFixed(1)} A${ri},${ri} 0 ${large} 0 ${xi1.toFixed(1)},${yi1.toFixed(1)} Z`;
    angle+=a;
    return `<path d="${d}" fill="${sl.color}" opacity=".9"/>`;
  }).join('');
  return `<svg width="${size}" height="${size}" viewBox="0 0 ${size} ${size}">${paths}</svg>`;
}
