# Bottle Inventory & QR Tracking Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a full bottle inventory system — FSL-coded TC culture vessels with QR labels, lifecycle tracking, and cross-links to notes/recipes/accessions/calendar/reminders.

**Architecture:** All code in `index.html`. Bottles stored in `bottles/data.enc` (same AES-256-GCM pattern as accessions/recipes). `🧪 Bottles` nav tab with drawer list panel + full main section. QR codes generated client-side via `qrcode.js` CDN. Labels print on 58mm thermal roll, B&W.

**Tech Stack:** Vanilla JS, existing `ghGet`/`ghPut`/`enc`/`dec` helpers, `qrcode.js` (CDN), CSS custom properties.

const BOT_STATUSES=['prepared','inoculated','growing','subcultured','contaminated','discarded'];
const BOT_STATUS_LABEL={prepared:'Prepared',inoculated:'Inoculated',growing:'Growing',subcultured:'Subcultured',contaminated:'Contaminated',discarded:'Discarded'};
const BOT_STATUS_NEXT={prepared:'inoculated',inoculated:'growing',growing:'subcultured'};
let _botFilter='all', _botEditingId=null, _botMode='single';

function _botNextCode(){
  const year=new Date().getFullYear();
  const nums=(S.bottles||[]).map(b=>parseInt(b.code?.split('-')[2])||0);
  const next=nums.length?Math.max(...nums)+1:1;
  return `FSL-${year}-${String(next).padStart(4,'0')}`;
}

function _botFmtTs(iso){
  if(!iso) return '';
  const d=new Date(iso);
  const dd=String(d.getDate()).padStart(2,'0');
  const mm=String(d.getMonth()+1).padStart(2,'0');
  const yy=String(d.getFullYear()).slice(2);
  const hh=String(d.getHours()).padStart(2,'0');
  const mi=String(d.getMinutes()).padStart(2,'0');
  return `${dd}/${mm}/${yy} ${hh}:${mi}`;
}

function _setBotFilter(f){
  _botFilter=f;
  document.querySelectorAll('.bot-filter-pill').forEach(p=>p.classList.toggle('active',p.dataset.bfilter===f));
  renderBottleList();
}

function renderBottleList(){
  const el=document.getElementById('bot-list');
  if(!el) return;
  const q=(document.getElementById('bot-search')?.value||'').toLowerCase();
  let list=(S.bottles||[]).filter(b=>{
    if(_botFilter!=='all'&&b.status!==_botFilter) return false;
    if(q) return (b.code||'').toLowerCase().includes(q)||(b.species||'').toLowerCase().includes(q)||(b.notes||'').toLowerCase().includes(q);
    return true;
  }).sort((a,b)=>new Date(b.date_prepared)-new Date(a.date_prepared));
  if(!list.length){ el.innerHTML=`<div style="padding:20px;text-align:center;color:var(--mu);font-size:13px">${q||_botFilter!=='all'?'No matching bottles.':'No bottles yet.'}</div>`; return; }
  el.innerHTML=list.map(b=>{
    const flairPills=(b.flair_ids||[]).map(fid=>{const f=(S.taxFlairs||[]).find(x=>x.id===fid);return f?`<span class="flair-pill ${f.color}" style="font-size:10px;padding:1px 5px">${esc(f.name)}</span>`:''}).join('');
    return `<div class="bot-item${S.curBot===b.id?' active':''}" onclick="openBottle('${b.id}')">
      <span class="bot-status-dot ${b.status||'prepared'}"></span>
      <div class="bot-item-info">
        <div class="bot-item-code">${esc(b.code)}</div>
        <div class="bot-item-sp">${esc(b.species||'—')}${b.stage?` · ${esc(b.stage)}`:''}</div>
        ${flairPills?`<div style="display:flex;flex-wrap:wrap;gap:2px;margin-top:2px">${flairPills}</div>`:''}
      </div>
    </div>`;
  }).join('');
}

  document.getElementById('bv-code').textContent=b.code;
