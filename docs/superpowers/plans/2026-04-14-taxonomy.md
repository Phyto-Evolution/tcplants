# Taxonomy Pack + Autocomplete + Flairs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add curated taxonomy packs (Carnivorous Plants, Orchids, Banana) to the TC Plants app with a species autocomplete dropdown, a pack browser in the Species section, and a flair system for labelling accession records.

**Architecture:** Taxonomy JSON files copied from PhytoCommerce into `taxonomy/` in the tcplants repo. At login, `loadTaxonomy()` fetches active packs and decrypts `taxonomy/custom.enc` (user-defined species + flairs), building `S.taxSpecies` and `S.taxFlairs` in memory. Autocomplete is a custom floating `<div>` wired to both species inputs. Flairs are stored on accession records as `flair_ids[]` + `flair_note`.

**Tech Stack:** Vanilla JS, CSS custom properties, existing `ghGet`/`ghPut`/`enc`/`dec` helpers, no new dependencies.

// ════════════════════════════════════════════════════════════
// TAXONOMY
// ════════════════════════════════════════════════════════════
async function loadTaxonomy(){
  // Determine active packs (default: all)
  let active;
  try{ active=new Set(JSON.parse(localStorage.getItem('tcplants_tax_packs')||'null')||TAX_ALL_IDS); }
  catch(e){ active=new Set(TAX_ALL_IDS); }

  // Fetch active pack JSONs in parallel
  const packIds=[...active].filter(id=>TAX_PACKS[id]);
  const fetched=await Promise.allSettled(
    packIds.map(id=>ghGet(`taxonomy/${TAX_PACKS[id]}/${id}.json`))
  );

  // Flatten genera[].species[] into S.taxSpecies
  S.taxSpecies=[];
  fetched.forEach((res,i)=>{
    if(res.status!=='fulfilled'||!res.value) return;
    const data=typeof res.value==='string'?JSON.parse(res.value):res.value;
    // ghGet returns the GitHub API response; content is base64 — decode it
    let pack;
    try{
      pack=typeof data.content==='string'
        ?JSON.parse(decodeURIComponent(escape(atob(data.content.replace(/\n/g,'')))))
        :data;
    }catch(e){ return; }
    for(const genus of (pack.genera||[])){
      for(const sp of (genus.species||[])){
        S.taxSpecies.push({
          full_name: sp.full_name||'',
          common_name: sp.common_name||'',
          conservation_status: sp.conservation_status||'',
          is_custom: false
        });
      }
    }
  });

  // Load custom.enc
  try{
    const f=await ghGet(TAX_CUSTOM_FILE);
    if(f){
      S.taxCustom=JSON.parse(await dec(ghDec(f.content)));
      S.taxCustomSha=f.sha;
    } else {
      S.taxCustom={species:[],flairs:[]};
      S.taxCustomSha=null;
    }
  }catch(e){
    S.taxCustom={species:[],flairs:[]};
    S.taxCustomSha=null;
  }

  // Append custom species
  for(const sp of (S.taxCustom.species||[])){
    S.taxSpecies.push({...sp, is_custom:true});
  }
  S.taxFlairs=S.taxCustom.flairs||[];
}

async function saveCustomTaxonomy(){
  const res=await ghPut(TAX_CUSTOM_FILE, await enc(JSON.stringify(S.taxCustom)), S.taxCustomSha,
    `lab: taxonomy custom — ${S.taxCustom.species.length} species, ${S.taxCustom.flairs.length} flairs`);
  S.taxCustomSha=res.content.sha;
}
```

function _taxConBadge(status){
  const map={
    critically_endangered:'CR',endangered:'EN',vulnerable:'VU',
    near_threatened:'NT',least_concern:'LC',data_deficient:'DD'
  };
  const cls={
    critically_endangered:'cr',endangered:'en',vulnerable:'vu',
    near_threatened:'nt',least_concern:'lc',data_deficient:'dd'
  };
  const code=map[status]; if(!code) return '';
  return `<span class="tax-badge ${cls[status]}">${code}</span>`;
}

let _taxActiveInput=null;
function _taxSuggest(input){
  _taxActiveInput=input;
  const q=(input.value||'').toLowerCase().trim();
  const dd=document.getElementById('tax-dropdown');
  if(!q||!S.taxSpecies.length){ dd.classList.remove('on'); return; }

  const matches=S.taxSpecies.filter(sp=>
    sp.full_name.toLowerCase().includes(q)||sp.common_name.toLowerCase().includes(q)
  );
  if(!matches.length){ dd.classList.remove('on'); return; }

  // Position dropdown under the input
  const rect=input.getBoundingClientRect();
  const parentRect=input.closest('.field').getBoundingClientRect();
  dd.style.width=input.offsetWidth+'px';
  dd.style.left=(rect.left-parentRect.left)+'px';
  // Attach to input's parent .field for stacking
  input.closest('.field').style.position='relative';
  input.closest('.field').appendChild(dd);

  dd.innerHTML=matches.map((sp,i)=>{
    // Find accessions of this species with flairs
    const accsWithFlairs=(S.accs||[]).filter(a=>
      (a.species||'').toLowerCase()===sp.full_name.toLowerCase()&&a.flair_ids?.length
    );
    const flairSubs=accsWithFlairs.map(a=>{
      const pills=a.flair_ids.map(fid=>{
        const f=S.taxFlairs.find(x=>x.id===fid);
        return f?`<span class="flair-pill ${f.color}">${esc(f.name)}</span>`:'';
      }).join('');
      return `<span style="font-size:11px;color:var(--mu)">${esc(a.number||a.species)}</span> ${pills}`;
    }).join(' · ');

    return `<div class="tax-row" data-idx="${i}" onclick="_taxDropdownSelect('${sp.full_name.replace(/'/g,"\\'")}')">
      <div class="tax-row-name">${esc(sp.full_name)}</div>
      <div class="tax-row-common">${esc(sp.common_name)}</div>
      ${_taxConBadge(sp.conservation_status)}
    </div>${accsWithFlairs.length?`<div class="tax-accs">${flairSubs}</div>`:''}`;
  }).join('');

  dd.classList.add('on');
}

function _taxDropdownSelect(name){
  if(_taxActiveInput) _taxActiveInput.value=name;
  document.getElementById('tax-dropdown').classList.remove('on');
  _taxActiveInput=null;
}

let _taxFocusIdx=-1;
function _taxDropdownKeyNav(e){
  const dd=document.getElementById('tax-dropdown');
  if(!dd.classList.contains('on')) return;
  const rows=dd.querySelectorAll('.tax-row');
  if(e.key==='ArrowDown'){ e.preventDefault(); _taxFocusIdx=Math.min(_taxFocusIdx+1,rows.length-1); }
  else if(e.key==='ArrowUp'){ e.preventDefault(); _taxFocusIdx=Math.max(_taxFocusIdx-1,0); }
  else if(e.key==='Enter'&&_taxFocusIdx>=0){ e.preventDefault(); rows[_taxFocusIdx].click(); _taxFocusIdx=-1; return; }
  else if(e.key==='Escape'){ dd.classList.remove('on'); _taxFocusIdx=-1; return; }
  rows.forEach((r,i)=>r.classList.toggle('focus',i===_taxFocusIdx));
}

// Close dropdown on outside click
document.addEventListener('click',function(e){
  if(!e.target.closest('#tax-dropdown')&&!e.target.matches('#ne-species,#ae-species')){
    document.getElementById('tax-dropdown')?.classList.remove('on');
    _taxFocusIdx=-1;
  }
});
```

function _renderGenusGroups(packIds){
  // Group S.taxSpecies by first word (genus) for packs in this category
  const catId=TAX_PACKS[packIds[0]];
  const species=S.taxSpecies.filter(sp=>!sp.is_custom);
  // We don't track which pack a species came from in memory, so group by genus (first word)
  const byGenus={};
  for(const sp of species){
    const genus=sp.full_name.split(' ')[0];
    if(!byGenus[genus]) byGenus[genus]=[];
    byGenus[genus].push(sp);
  }
  if(!Object.keys(byGenus).length) return '<p style="font-size:13px;color:var(--mu)">No species loaded. Enable this pack above.</p>';
  return Object.entries(byGenus).slice(0,50).map(([genus,sps])=>`
    <div class="tax-genus">
      <div class="tax-genus-name">${esc(genus)} <span style="font-weight:400;color:var(--mu)">(${sps.length})</span></div>
      ${sps.map(sp=>`<div class="tax-sp-row">
        <span class="tax-sp-sci">${esc(sp.full_name)}</span>
        <span class="tax-sp-com">${esc(sp.common_name)}</span>
        ${_taxConBadge(sp.conservation_status)}
      </div>`).join('')}
    </div>`).join('');
}

function _taxToggleCat(catId,enabled){
  const active=_getActivePacks();
  const catPacks=Object.entries(TAX_PACKS).filter(([,v])=>v===catId).map(([k])=>k);
  catPacks.forEach(p=>enabled?active.add(p):active.delete(p));

  toast(enabled?'Pack enabled — reload to update autocomplete':'Pack disabled — reload to update autocomplete','info');
}
